variable
  X Y Z : Set

record Container : Set₁ where
  constructor _◃_
  field Sh   : Set
        Pos  : Sh → Set

⟦_⟧c : Container → Set → Set
⟦ Sh ◃ Pos ⟧c X = Σ Sh λ sh → Pos sh → X


ConFun : (C : Container) → (f : X → Y) → ⟦ C ⟧c X → ⟦ C ⟧c Y
ConFun C f (shC , posC) = shC , (f ∘ posC)

functor-id : (C : Container) → ConFun {X = X} C id ≡ id
functor-id C = refl

functor-comp : (C : Container) → (f : X → Y) → (g : Y → Z)
             → ConFun C (g ∘ f) ≡ ConFun C g ∘ ConFun C f
functor-comp C f g = refl

ListC : Container
List : (A : Set) → Set

ListC = Nat ◃ Fin
List A = ⟦ ListC ⟧c A

record IxCon (M : Set) : Set₁ where
  constructor _◃_
  field Sh   : Set
        Pos  : Sh → M → Set

⟦_⟧ : {M : Set} → IxCon M → (M → Set) → M → Set
⟦_⟧ (Sh ◃ Pos) X m = Σ Sh λ sh → Pos sh m → X m

record ExtCon (M : UE SET) : Set where
  field Sh   : UE SET
        Pos  : ⟦ Sh ⟧UE → FinFam M

  Pos′ : ⟦ Sh ⟧UE → PowE M
  Pos′ s = Pow (Pos s)
open ExtCon public

EqExtCon : {M : UE SET} → (C D : ExtCon M) → UE PROP
EqExtCon {M} record { Sh = ShC ; Pos = PosC } record { Sh = ShD ; Pos = PosD }
  = Π₂ ShC ShD (\ sC sD → EqFinFam (PosC sC) (PosD sD))


module _ {M : UE SET} where

  EmptyCon : ExtCon M
  Sh EmptyCon = ‵Prf ‵0
  Pos EmptyCon ()

  IdCon : ⟦ M ⟧UE →  ExtCon M
  Sh (IdCon m) = ‵Prf ‵1
  Pos (IdCon m) _ = Fam1 m

  _+C_ : (C D : ExtCon M) → ExtCon M
  Sh (C +C D) = Sh C +E Sh D
  Pos (C +C D) (ff , shC) = Pos C shC
  Pos (C +C D) (tt , shD) = Pos D shD

  _×C_ : (C D : ExtCon M) → ExtCon M
  Sh (C ×C D) = Sh C ×E Sh D
  Pos (C ×C D) (shC , shD)
    = Pos C shC +F Pos D shD

module _ {M : UE SET}(C : ExtCon M)(X : PowE M) where

-- pairs of shapes and functions from positions
  ⟦_⟧C : UE SET
  ⟦_⟧C = ‵Σ (Sh C) λ s → [ Pos′ C s →i X ]

  ⟦_⟧CE : Set
  ⟦_⟧CE = ⟦ ⟦_⟧C ⟧UE

-- I don′t really use the actual set of positions of C, only to compute the respective index
-- the content of (Ix (Pos C S)) C does not change the composition here

-- "composition" of containers
-- have a C and a family of Ds and combine them (such that at each position of C there′s a D container)
-- C provides the "template", how to combine the Ds into one

_-◃-_ : {M N : UE SET} → (C : ExtCon M) → (⟦ M ⟧UE → ExtCon N) → ExtCon N
Sh  (C -◃- Ds) = ⟦ C ⟧C λ m → Sh (Ds m) -- a shape of C and for each position (Pos′ C s), a shape in D
Pos (_-◃-_ {I} C Ds) (s , k)
  = ΣFam (Ix (Pos C s))
         λ p → Pos (Ds (ix (Pos C s) p)) (k (ix (Pos C s) p) (p , EqR I (ix (Pos C s) p)))

mapC : {M : UE SET} → (C : ExtCon M)
     → {X Y : PowE M} → ⟦ [ X →i Y ] ⟧UE
     → ⟦ ⟦ C ⟧C X →E ⟦ C ⟧C Y ⟧UE
mapC C f (s , k)  = s , λ m p → f m (k m p)

mapCid : {M : UE SET} {X : PowE M} → (C : ExtCon M)
      → Eq′ (⟦ C ⟧C X →E ⟦ C ⟧C X) _ (mapC C (λ _ → id)) id
mapCid C = EqR′

mapCcomp : {M : UE SET} → (C : ExtCon M)
        → {X Y Z : PowE M} → (f : ⟦ [ X →i Y ] ⟧UE) → (g : ⟦ [ Y →i Z ] ⟧UE)
        → Eq′ (⟦ C ⟧C X →E ⟦ C ⟧C Z) _
              (mapC C (λ m → f m - g m))
              (mapC C f - mapC C g)
mapCcomp C f g = EqR′

-- ## natural transformations ##

_⇒_ : {M : UE SET} → (C D : ExtCon M) → UE SET
C ⇒ D = ‵Π (Sh C) λ s → ⟦ D ⟧C (Pos′ C s)

idNT : {M : UE SET} → (C : ExtCon M) → ⟦ C ⇒ C ⟧UE
fst (idNT C s) = s
snd (idNT C s) m p = p

composeNT : {M : UE SET} → (C D E : ExtCon M)
          → ⟦ C ⇒ D ⟧UE → ⟦ D ⇒ E ⟧UE → ⟦ C ⇒ E ⟧UE
fst (composeNT C D E f g sc) = fst (g (fst (f sc)))
snd (composeNT C D E f g sc) m pe with f sc
... | sd , kd with g sd
... | se , ke = kd m (ke m pe)

_-nt-_ : {M : UE SET} → {C D E : ExtCon M}
          → ⟦ C ⇒ D ⟧UE → ⟦ D ⇒ E ⟧UE → ⟦ C ⇒ E ⟧UE
_-nt-_ {C = C} {D} {E} = composeNT C D E
infixr 20 _-nt-_

appNT : {M : UE SET} → (C D : ExtCon M) → (nt : ⟦ C ⇒ D ⟧UE) → {X : PowE M}
      → ⟦ ⟦ C ⟧C X →E ⟦ D ⟧C X ⟧UE
appNT C D nt (s , k) = mapC D k (nt s)

naturality : {M : UE SET} → (C D : ExtCon M) → (nt : ⟦ C ⇒ D ⟧UE)
           → {X Y : PowE M} → (f : ⟦ [ X →i Y ] ⟧UE)
           → let g = ⟦ C ⟧C X →E ⟦ D ⟧C Y in
             Eq′ g g (mapC C f - appNT C D nt) (appNT C D nt - mapC D f)
naturality C D nt f = EqR′

-- projection is natural:

projLeftN : {M : UE SET} → (C D : ExtCon M) → ⟦ (C ×C D) ⇒ C ⟧UE
fst (projLeftN C D (sC , sD)) = sC
snd (projLeftN C D (sC , sD)) m (pos , q) = (ff , pos) , q

projRightN : {M : UE SET} → (C D : ExtCon M) → ⟦ (C ×C D) ⇒ D ⟧UE
fst (projRightN C D (sC , sD)) = sD
snd (projRightN C D (sC , sD)) m (pos , q) = (tt , pos) , q

mapO : {M N : UE SET} (C D : ExtCon M)(Es : ⟦ M ⟧UE → ExtCon N)
      → ⟦ C ⇒ D ⟧UE → ⟦ (C -◃- Es) ⇒ (D -◃- Es) ⟧UE
fst (mapO C D Es f (shC , shE))
    = let (shD , fposD) = f shC in shD , λ m posD → shE m (fposD m posD)
snd (mapO {M} {N} C D Es f (shC , shE)) n ((posD , posE) , q)
    = let (shD , k)    = f shC
          (posC , p′)  = k (ix (Pos D shD) posD) (posD , EqR _ _)
          (posE′ , q′)
            = J M (ix (Pos C shC) posC) (ix (Pos D shD) posD) p′
               (λ y p → (‵Σ ⟦ Ix (Pos (Es y) (shE y (posC , p))) ⟧F λ pos → ‵Prf (Eq N N (ix (Pos (Es y) (shE y (posC , p))) pos) n))
                      →E ‵Σ ⟦ Ix (Pos (Es (ix (Pos C shC) posC)) (shE (ix (Pos C shC) posC) (posC , EqR M (ix (Pos C shC) posC)))) ⟧F
                          λ pos → ‵Prf (Eq N N (ix (Pos (Es (ix (Pos C shC) posC)) (shE (ix (Pos C shC) posC) (posC , EqR M (ix (Pos C shC) posC)))) pos) n))
                id (posE , q)
      in (posC , posE′) , q′

mapNT : {M N : UE SET} (C : ExtCon M) (Ds Es : ⟦ M ⟧UE → ExtCon N)
        → ⟦ ‵Π M (λ m → Ds m ⇒ Es m) ⟧UE → ⟦ (C -◃- Ds) ⇒ (C -◃- Es) ⟧UE
fst (fst (mapNT C Ds Es f (shC , _))) = shC
snd (fst (mapNT C Ds Es f (shC , fshD))) m posC = fst (f m (fshD m posC))
snd (mapNT {M} C Ds Es f (shC , fshD)) n ((posC , posE) , q)
    = let m  = ix (Pos C shC) posC
          shD = fshD (ix (Pos C shC) posC) (posC , EqR M (ix (Pos C shC) posC))
          shE , fposE = f m shD
          posE′ , q′ = fposE n (posE , q)
      in (posC , posE′) , q′

-- ## deriatives

-- partial is a container with a hole at i
Partial : {M : UE SET} → ExtCon M → (m : ⟦ M ⟧UE) → ExtCon M
Sh  (Partial C m) = ‵Σ (Sh C) λ sh → Pos′ C sh m
Pos (Partial C m) (sh , pos , q) = (Pos C sh) ─F pos

Pointed′ : {M : UE SET} → ExtCon M → (m : ⟦ M ⟧UE) → ExtCon M
Pointed′ C m = IdCon m ×C Partial C m

Pointed : {M : UE SET} → ExtCon M → (m : ⟦ M ⟧UE) → ExtCon M
Sh (Pointed C m) = ‵Σ (Sh C) λ s → Pos′ C s m
Pos (Pointed C m) (s , p , q) = Pos C s

mapNTId : {M : UE SET} (C : ExtCon M) (Ds : ⟦ M ⟧UE → ExtCon M) (m : ⟦ M ⟧UE)
        → Eq′ ((Pointed C m -◃- Ds) ⇒ (Pointed C m -◃- Ds)) ((Pointed C m -◃- Ds) ⇒ (Pointed C m -◃- Ds))
              (mapNT (Pointed C m) Ds Ds (\ _ s → s , (λ _ p → p)))
              λ s → s , (λ _ p → p)
mapNTId C Ds m = EqR′

composeΠ : {M : UE SET} → {Ds Es Fs : ⟦ M ⟧UE → ExtCon M}
         → (f : ⟦ ‵Π M (λ m → Ds m ⇒ Es m) ⟧UE) → (g : ⟦ ‵Π M (λ m → Es m ⇒ Fs m) ⟧UE)
         → ⟦ ‵Π M (λ m → Ds m ⇒ Fs m) ⟧UE
composeΠ f g m shD with f m shD
... | (shE , posE) with g m shE
... | (shF , posF) = shF , λ n p → posE n (posF n p)

mapNTcomp : {M : UE SET} (C : ExtCon M) (Ds Es Fs : ⟦ M ⟧UE → ExtCon M) (m : ⟦ M ⟧UE)
          → (f : ⟦ ‵Π M (λ m → Ds m ⇒ Es m) ⟧UE) → (g : ⟦ ‵Π M (λ m → Es m ⇒ Fs m) ⟧UE)
          → Eq′ ((Pointed C m -◃- Ds) ⇒ (Pointed C m -◃- Fs)) ((Pointed C m -◃- Ds) ⇒ (Pointed C m -◃- Fs))
                (composeNT (Pointed C m -◃- Ds) (Pointed C m -◃- Es)
                   (Pointed C m -◃- Fs) (mapNT (Pointed C m)  Ds Es f) (mapNT (Pointed C m) Es Fs g))
                (mapNT (Pointed C m) Ds Fs (composeΠ f g))
mapNTcomp C Ds Es Fs m f g = EqR′


module _ (M : UE SET)(C : ExtCon M) (m : ⟦ M ⟧UE) where

  F : (⟦ M ⟧UE → ExtCon M) → ExtCon M
  F = Pointed C m -◃-_

  FId : (Ds : ⟦ M ⟧UE → ExtCon M)
        → Eq′ (F Ds ⇒ F Ds) (F Ds ⇒ F Ds) (mapNT (Pointed C m) Ds Ds (\ _ s → s , (λ _ p → p)))
              λ s → s , (λ _ p → p)
  FId Ds = EqR′

  Fcomp : (Ds Es Fs : ⟦ M ⟧UE → ExtCon M) → (f : ⟦ ‵Π M (λ m → Ds m ⇒ Es m) ⟧UE) → (g : ⟦ ‵Π M (λ m → Es m ⇒ Fs m) ⟧UE)
        → Eq′ (F Ds ⇒ F Fs) (F Ds ⇒ F Fs) (composeNT (Pointed C m -◃- Ds) (Pointed C m -◃- Es)
                   (Pointed C m -◃- Fs) (mapNT (Pointed C m)  Ds Es f) (mapNT (Pointed C m) Es Fs g))
                (mapNT (Pointed C m) Ds Fs (composeΠ f g))
  Fcomp Ds Es Fs f g = EqR′

module _ {M : UE SET} (C : ExtCon M) where

  -- I can fill a container with identity containers
  Cid< : ⟦ C ⇒ (C -◃- IdCon) ⟧UE
  fst (Cid< s) = s , λ _ _ → _
  snd (Cid< s) n ((p , _) , q) = p , q

  -- but still can recover the original one
  Cid : ⟦ (C -◃- IdCon) ⇒ C ⟧UE
  fst (Cid (s , sid)) = s
  snd (Cid s) n (p , q) = (p , hide ⟨⟩) , q


  CidEq : Eq′ (C ⇒ C) (C ⇒ C)
        (composeNT C (C -◃- IdCon) C Cid< Cid)
        (idNT C)
  CidEq = EqR′

  CidEq< : Eq′ ((C -◃- IdCon) ⇒ (C -◃- IdCon)) ((C -◃- IdCon) ⇒ (C -◃- IdCon))
         (composeNT (C -◃- IdCon) C (C -◃- IdCon) Cid Cid<)
         (idNT (C -◃- IdCon))
  CidEq< = EqR′

  -- insert a copy of the original container at each of its positions
  copy : ⟦ C ⇒ (C -◃- (λ _ → C)) ⟧UE
  fst (copy s) = s , λ _ _ → s
  snd (copy s) m ((p , p′) , q) = p′ , q





module _ {M : UE SET} (C : ExtCon M) where

  -- return the focus/the context

  here : (m : ⟦ M ⟧UE) → ⟦ Pointed C m ⇒ IdCon m ⟧UE
  fst (here m (sh , p , q)) = hide ⟨⟩
  snd (here m (sh , p , q)) n (p′ , q′) = p , J M m n q′ (\k _ → ‵Prf (Eq M M (ix (Pos C sh) p) k)) q

  context : (m : ⟦ M ⟧UE) → ⟦ Pointed C m ⇒ Partial C m ⟧UE
  fst (context m sh) = sh
  snd (context m sh) n ((p , p′) , q) = p , q


module _ {M : UE SET}(Ds : ⟦ M ⟧UE → ExtCon M) (m : ⟦ M ⟧UE) where

  -- if the outer container is the identity, we can just look at the inner container
  outerId : ⟦ (IdCon m -◃- Ds) ⇒ Ds m ⟧UE
  fst (outerId (_ , s)) = s m ((hide ⟨⟩) , (EqR M m))
  snd (outerId (_ , s)) n (s' , p') = (_ , s') , p'

module _ {M : UE SET} (C : ExtCon M) (Ds : ⟦ M ⟧UE → ExtCon M) (m : ⟦ M ⟧UE) where

  -- forget about outer structure, and return the container that is stored at the focus
  -- maybe this should be called flattenO
  here′ : ⟦ (Pointed C m -◃- Ds) ⇒ Ds m ⟧UE
  here′ = composeNT (Pointed C m -◃- Ds) (IdCon m -◃- Ds) (Ds m)
            (mapO (Pointed C m) (IdCon m) Ds (here C m))
            (outerId Ds m)

  extract : ⟦ F M C m Ds ⇒ Ds m ⟧UE
  extract = here′

module _ {M : UE SET} (C : ExtCon M) (D : ExtCon M) where

  -- flattens the inner structure and only remembers the position in focus
  -- maybe this should be called flattenI
  flatten : ⟦ (C -◃- Pointed D) ⇒ C ⟧UE
  flatten  = composeNT (C -◃- Pointed D) (C -◃- IdCon) C
                 (mapNT C (Pointed D) IdCon (here D))
                 (Cid C)

module _ {M : UE SET} (C : ExtCon M) where

  flattenC : ⟦ (C -◃- Pointed C) ⇒ C ⟧UE
  flattenC = flatten C C

  flattenPartial : (m : ⟦ M ⟧UE) → ⟦ (Partial C m -◃- Pointed C) ⇒ Partial C m ⟧UE
  flattenPartial m = flatten (Partial C m) C

  flattenPointed : (m : ⟦ M ⟧UE) → ⟦ (Pointed C m -◃- Pointed C) ⇒ Pointed C m ⟧UE
  flattenPointed m = flatten (Pointed C m) C


module _ {M N : UE SET} (C : ExtCon M) (D : ⟦ M ⟧UE → ExtCon N) where

  -- remove the focussed position from the outer structure

  rmFocus : (m : ⟦ M ⟧UE)  → ⟦ (Pointed C m -◃- D) ⇒ (Partial C m -◃- D) ⟧UE
  rmFocus m = mapO (Pointed C m) (Partial C m) D (context C m)

  -- forgetting the focus is the operation "up"


module _ {M : UE SET} (C : ExtCon M) where

  -- up forgets about the focus
  up : (m : ⟦ M ⟧UE) → ⟦ Pointed C m ⇒ C ⟧UE
  fst (up m (sh , p , q)) = sh
  snd (up m (sh , p , q)) n (p′ , q′) = p′ , q′

  down : ⟦ C ⇒ (C -◃- Pointed C) ⟧UE
  fst (fst (down s)) = s
  snd (fst (down s)) m p = s , p
  snd (down s) m ((p , p′) , q) = p′ , q


copyPointed : {M : UE SET} (C : ExtCon M) (m : ⟦ M ⟧UE)
