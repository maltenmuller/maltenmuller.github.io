data Sort : Set where
  SET PROP : Sort

data UE : Sort → Set
⟦_⟧UE : {s : Sort} → UE s → Set

data UE where
  ‵0 ‵1  : UE PROP
  ‵2     : UE SET
  ‵Σ     : {s : Sort} → (S : UE s)    → (T : ⟦ S ⟧UE → UE s) → UE s
  ‵Π     : {s : Sort} → (S : UE SET)  → (T : ⟦ S ⟧UE → UE s) → UE s
  ‵Prf   : UE PROP → UE SET

⟦ ‵0 ⟧UE = Zero
⟦ ‵1 ⟧UE = One
⟦ ‵2 ⟧UE = Two
⟦ ‵Σ S T ⟧UE = Σ ⟦ S ⟧UE λ s → ⟦ T s ⟧UE
⟦ ‵Π S T ⟧UE = (s : ⟦ S ⟧UE) → ⟦ T s ⟧UE
⟦ ‵Prf P ⟧UE = Hide ⟦ P ⟧UE

⟦_⟧P : UE PROP → Set
⟦ P ⟧P = ⟦ ‵Prf P ⟧UE

_→E_ : UE SET → UE SET → UE SET
S →E T = ‵Π S λ _ → T

_+E_ : UE SET → UE SET → UE SET
S +E T = ‵Σ ‵2 λ { ff → S ; tt → T}

_×E_ :  UE SET → UE SET → UE SET
S ×E T = ‵Σ S λ _ → T

_&P_ : UE PROP → UE PROP → UE PROP
S &P T = ‵Σ S λ _ → T

_→P_ : {x : Sort} → UE PROP → UE x → UE x
S →P T = ‵Π (‵Prf S) λ _ → T

PowE : UE SET → Set
PowE I = ⟦ I ⟧UE → UE SET

So : Two → UE PROP
So ff = ‵0
So tt = ‵1

No : Two → UE PROP
No b = So (not b)

postulate
  proof : {P : UE PROP} → ⟦ ‵Prf P ⟧UE → ⟦ P ⟧UE

-- term equality
-- if the types are equal so are the values
Eq : (A B : UE SET) → ⟦ A ⟧UE → ⟦ B ⟧UE → UE PROP

-- heterogeneous pi types
Π₂ : {x : Sort} (A B : UE SET) → (F : ⟦ A ⟧UE → ⟦ B ⟧UE → UE x) → UE x
Π₂ A B F = ‵Π A λ a → ‵Π B λ b → Eq A B a b →P F a b

Eq ‵2 ‵2 ff ff = ‵1
Eq ‵2 ‵2 ff tt = ‵0
Eq ‵2 ‵2 tt ff = ‵0
Eq ‵2 ‵2 tt tt = ‵1
Eq (‵Σ S₀ T₀) (‵Σ S₁ T₁) (s₀ , t₀) (s₁ , t₁) = Eq S₀ S₁ s₀ s₁ &P Eq (T₀ s₀) (T₁ s₁) t₀ t₁
Eq (‵Π S₀ T₀) (‵Π S₁ T₁) f₀ f₁ = Π₂ S₀ S₁ (λ s₀ s₁ → Eq (T₀ s₀) (T₁ s₁) (f₀ s₀) (f₁ s₁))
Eq _ _ _ _ = ‵1

--^^ types are different, or propositions

-- equality on types
_⇔_ : UE SET → UE SET → UE PROP
‵2 ⇔ ‵2 = ‵1
‵Σ S₀ T₀  ⇔ ‵Σ S₁ T₁  = (S₀ ⇔ S₁) &P Π₂ S₀ S₁ (λ s₀ s₁ → T₀ s₀ ⇔ T₁ s₁)
‵Π S₀ T₀  ⇔ ‵Π S₁ T₁  = (S₁ ⇔ S₀) &P Π₂ S₁ S₀ (λ s₁ s₀ → T₀ s₀ ⇔ T₁ s₁)
‵Prf S    ⇔ ‵Prf T    = (S →P T) &P (T →P S)
_ ⇔ _ = ‵0

-- wrapping Eq into a layer of ⟦ ‵Prf _ ⟧UE
record Eq′ (A B : UE SET)(a : ⟦ A ⟧UE)(b : ⟦ B ⟧UE) : Set where
  constructor eq
  field eqprf : ⟦ Eq A B a b ⟧P
open Eq′ public

eqsg : {S0 S1 : UE SET}
  {T0 : ⟦ S0 ⟧UE → UE SET} {T1 : ⟦ S1 ⟧UE → UE SET}
  {s0 : ⟦ S0 ⟧UE}{s1 : ⟦ S1 ⟧UE }{t0 : ⟦ T0 s0 ⟧UE}{t1 : ⟦ T1 s1 ⟧UE}
   → Eq′ S0 S1 s0 s1 → Eq′ (T0 s0) (T1 s1) t0 t1
   → Eq′ (‵Σ S0 T0) (‵Σ S1 T1) (s0 , t0) (s1 , t1)
eqsg (eq (hide qs)) (eq (hide qt)) = eq (hide (qs , qt))

-- when are two things apart
-- types are equal and values are not
Different : (A B : UE SET) → ⟦ A ⟧UE → ⟦ B ⟧UE → UE SET
Different ‵2 ‵2 ff ff = ‵Prf ‵0
Different ‵2 ‵2 ff tt = ‵Prf ‵1
Different ‵2 ‵2 tt ff = ‵Prf ‵1
Different ‵2 ‵2 tt tt = ‵Prf ‵0
Different (‵Σ S T) (‵Σ S′ T′) (s , t) (s′ , t′) = Different S S′ s s′ +E Different (T s) (T′ s′) t t′
Different (‵Π S T) (‵Π S′ T′) f g
  = ‵Σ S λ s → ‵Σ S′ λ s′ → ‵Σ (‵Prf (Eq S S′ s s′)) λ _ → Different (T s) (T′ s′) (f s) (g s′)
Different _ _ _ _ = ‵Prf ‵0                 -- types are different, or propositional


-- if two things are both the same and different...
magic : (A B : UE SET) → (a : ⟦ A ⟧UE) → (b : ⟦ B ⟧UE)
       → ⟦ Eq A B a b ⟧P → ⟦ Different A B a b ⟧UE → Zero
magic ‵2 ‵2 ff ff p ()
magic ‵2 ‵2 ff tt () p
magic ‵2 ‵2 tt ff () p
magic ‵2 ‵2 tt tt (hide x) ()
magic (‵Σ S T) (‵Σ S′ T′) (s , t) (s′ , t′) (hide q) (ff , p)
 = magic S S′ s s′ (hide (fst q)) p
magic (‵Σ S T) (‵Σ S′ T′) (s , t) (s′ , t′) (hide q) (tt , p)
  = magic (T s) (T′ s′) t t′ (hide (snd q)) p
magic (‵Π S T) (‵Π S′ T′) f g (hide q) (s , s′ , q′ , p)
  = magic (T s) (T′ s′) (f s) (g s′) (hide (q s s′ q′)) p

-- coercion and coherence

-- coercion
-- if I hae a proof that two types are equal,
-- then I can transform an element of one into an element of the other

-- coherence
  -- if I have a proof that two types are equal,
  -- I can get an equality proof between an element of one type and the same element but coerced to the other type


coe : (A B : UE SET) → ⟦ ‵Prf (A ⇔ B) ⟧UE → ⟦ A ⟧UE → ⟦ B ⟧UE
postulate coh : (A B : UE SET) → (Q : ⟦ ‵Prf (A ⇔ B) ⟧UE) → (a : ⟦ A ⟧UE)
              → ⟦ ‵Prf (Eq A B a (coe A B Q a)) ⟧UE

coe ‵2 ‵2 Q a = a
coe (‵Σ S0 T0) (‵Σ S1 T1) (hide q) (s0 , t0)
  = let s1 = (coe S0 S1 (hide (fst q)) s0)
    in s1 , coe (T0 s0) (T1 s1) (hide ((snd q) s0 s1 (coh S0 S1 (hide (fst q)) s0))) t0
coe (‵Π S0 T0) (‵Π S1 T1) (hide q) f0 s1
  = let s0 = coe S1 S0 (hide (fst q)) s1
    in coe (T0 s0) (T1 s1) (hide (snd q s1 s0 (coh S1 S0 (hide (fst q)) s1))) (f0 s0)
coe (‵Prf A) (‵Prf B) (hide q) a = hide ((fst q) a)

postulate
  Resp : (X : UE SET) → (x y : ⟦ X ⟧UE) → (q : ⟦ ‵Prf (Eq X X x y) ⟧UE)
       → (P : ⟦ X ⟧UE → UE SET)
       → ⟦ ‵Prf (P x ⇔ P y) ⟧UE

postulate  EqR : (A : UE SET) → (a : ⟦ A ⟧UE) → ⟦ ‵Prf (Eq A A a a) ⟧UE

EqR′ : {S : UE SET} → {s : ⟦ S ⟧UE} → Eq′ S S s s
eqprf (EqR′ {S} {s}) = EqR S s

-- coe and Resp together give transport of values across paths
-- (for example in the J-rule:)

J : (A : UE SET) → (a b : ⟦ A ⟧UE) → (q : ⟦ ‵Prf (Eq A A a b) ⟧UE)
  → (P : (b : ⟦ A ⟧UE) → ⟦ ‵Prf (Eq A A a b) ⟧UE → UE SET)
  → ⟦ P a (EqR A a) ⟧UE → ⟦ P b q ⟧UE
J X x y Q@(hide q) P p
  = coe (P x (EqR X x)) (P y Q)
    (Resp (‵Σ X λ y → ‵Prf (Eq X X x y)) (x , EqR X x) (y , Q) (hide (q , ⟨⟩)) λ (y , q) → P y q)
    p

J′ : {X : UE SET} → {x y : ⟦ X ⟧UE} → (q : Eq′ X X x y)
  → (P : (y : ⟦ X ⟧UE) → Eq′ X X x y → UE SET)
  → ⟦ P x EqR′ ⟧UE → ⟦ P y q ⟧UE
J′ {X} {x} {y} (eq q) P p = J X x y q (λ y q → P y (eq q)) p


-- Eq is symmetric
EqS : (A B : UE SET) → (a : ⟦ A ⟧UE)(b : ⟦ B ⟧UE)
        → ⟦ Eq A B a b ⟧UE → ⟦ Eq B A b a ⟧UE
EqS ‵2 ‵2 ff ff q = ⟨⟩
EqS ‵2 ‵2 tt tt q = ⟨⟩
EqS ‵2 (‵Σ B T) a b q = ⟨⟩

EqS ‵2 (‵Prf B) a b q = ⟨⟩
EqS (‵Σ A T) ‵2 a b q = ⟨⟩
EqS (‵Σ A A′) (‵Σ B B′) (a , a′) (b , b′) (q , q′)
  = EqS A B a b q , EqS (A′ a) (B′ b) a′ b′ q′

EqSh : {A B : UE SET} → {a : ⟦ A ⟧UE}{b : ⟦ B ⟧UE}
        → Hide ⟦ Eq A B a b ⟧UE → Hide ⟦ Eq B A b a ⟧UE
EqSh {A} {B} {a} {b} (hide q) = hide (EqS A B a b q)

Eqhelper : forall {A B a b}
         → ⟦ Eq A B a b ⟧UE → Eq′ A B a b
expose (eqprf (Eqhelper q)) = q

Eqhelper′ : forall {A B a b}
          → Eq′ A B a b → Hide ⟦ Eq A B a b ⟧UE
Eqhelper′ (eq (hide q)) = hide (irrAx q)

EQR : (A : UE SET) → ⟦ ‵Prf (A ⇔ A) ⟧UE
expose (EQR ‵2) = ⟨⟩
fst (expose (EQR (‵Σ A T))) = unhide (EQR A)
snd (expose (EQR (‵Σ A B))) a a' q = unhide (J A a a' q (\ a' _ → ‵Prf ((B a) ⇔ (B a'))) (EQR (B a)))
fst (expose (EQR (‵Π A T))) = unhide (EQR A)
snd (expose (EQR (‵Π A B))) a a' q = unhide (J A a a' q (\ a' _ → ‵Prf ((B a') ⇔ (B a))) (EQR (B a)))
expose (EQR (‵Prf A)) = unhide , unhide


applyFun : {A : UE SET} {B : ⟦ A ⟧UE → UE SET} → (f f′ : ⟦ ‵Π A B ⟧UE)

         → Eq′ (B a) (B a) (f a) (f′ a)
expose (eqprf (applyFun {A} f f′ (eq (hide qf)) a)) = qf a a (EqR A a)

extri : (S : UE SET) (P : ⟦ S ⟧UE → ⟦ S ⟧UE → UE PROP)
      → ((s : ⟦ S ⟧UE) → ⟦ P s s ⟧P)
      → ⟦ Π₂ S S P ⟧P
expose (extri S P q) s s′ q′ = proof (J S s s′ q′ (λ s′ _ → ‵Prf (P s s′)) (q s))

ext : (S : UE SET) (T0 T1 : ⟦ S ⟧UE → UE SET)
    → (f0 : ⟦ ‵Π S T0 ⟧UE) → (f1 : ⟦ ‵Π S T1 ⟧UE)
    → ((s : ⟦ S ⟧UE) → Eq′ (T0 s) (T1 s) (f0 s) (f1 s))
    → Eq′ (‵Π S T0) (‵Π S T1) f0 f1
eqprf (ext S T0 T1 f0 f1 q) = extri S _ λ s → eqprf (q s)

_=$$=_ : forall {S T : UE SET}
  {f g : ⟦ ‵Π S (λ _ → T) ⟧UE } -> Eq′ (‵Π S (λ _ → T)) (‵Π S (λ _ → T)) f g ->
  {x y : ⟦ S ⟧UE} -> Eq′ S S x y -> Eq′ T T (f x) (g y)
eqprf (_=$$=_ {S} {T} {f} {g} qf {x} {y} qx)
  = J′ qf (λ g eq → ‵Prf (Eq T T (f x) (g y)))
   (J′ qx (λ y eq → ‵Prf (Eq T T (f x) (f y))) (EqR T (f x)))

module _ {I : UE SET} where
-- working with indexed sets and functions

  [_] : PowE I → UE SET
  [_] X = ‵Π I X

  _→i_ : (X Y : PowE I) → PowE I
  (X →i Y) i = X i →E Y i

  <_> : PowE I → UE SET
  <_> X = ‵Σ I X

  _×i_ : (X Y : PowE I) → PowE I
  (X ×i Y) i = X i ×E Y i
