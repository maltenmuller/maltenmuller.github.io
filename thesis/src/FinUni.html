data UF : Set
⟦_⟧F : UF → UE SET
⟦_⟧FE : UF → Set

⟦ S ⟧FE = ⟦ ⟦ S ⟧F ⟧UE

-- the universe of finite sets
-- take 0,1,2 and close under sg
data UF where
  so  : Two → UF
  ‵2  : UF
  ‵Σ  : (S : UF) (T : ⟦ S ⟧FE → UF) → UF

⟦ so b ⟧F   = ‵Prf (So b)
⟦ ‵2 ⟧F    = ‵2
⟦ ‵Σ S T ⟧F = ‵Σ ⟦ S ⟧F λ s → ⟦ T s ⟧F

-- elements of a finite universe have decidable equality
DecUF : (A B : UF) → (a : ⟦ A ⟧FE) → (b : ⟦ B ⟧FE)
         → ⟦ Different ⟦ A ⟧F ⟦ B ⟧F a b +E ‵Prf (Eq ⟦ A ⟧F ⟦ B ⟧F a b) ⟧UE
DecUF (so tt) (so tt) a b = tt , _
DecUF (so x) ‵2 a b = tt , _
DecUF (so x) (‵Σ B T) a b = tt , _
DecUF ‵2 (so x) a b = tt , _
DecUF ‵2 ‵2 ff ff = tt , _
DecUF ‵2 ‵2 ff tt = ff , _
DecUF ‵2 ‵2 tt ff = ff , _
DecUF ‵2 ‵2 tt tt = tt , _
DecUF ‵2 (‵Σ B T) a b = tt , _
DecUF (‵Σ A T) (so x) a b = tt , _
DecUF (‵Σ A T) ‵2 a b = tt , _
DecUF (‵Σ A B) (‵Σ A′ B′) (a , b) (a′ , b′) with DecUF A A′ a a′
... | ff , le = ff , (ff , le)
... | tt , (hide le) with DecUF (B a) (B′ a′) b b′
... | ff , ri = ff , tt , ri
... | tt , (hide ri) = tt , hide (le , ri)

-- homogeneous version of dec equality
DecUFh : (A : UF) → (a b : ⟦ A ⟧FE)
      → ⟦ Different ⟦ A ⟧F ⟦ A ⟧F a b +E ‵Prf (Eq ⟦ A ⟧F ⟦ A ⟧F a b) ⟧UE
DecUFh A a b = DecUF A A a b

conv : (A B : UF) → (a : ⟦ A ⟧FE) → (b : ⟦ B ⟧FE)
     →  ⟦ Different ⟦ A ⟧F ⟦ B ⟧F a b ⟧UE → ⟦ So (not (fst (DecUF A B a b))) ⟧UE
conv ‵2 ‵2 ff tt df = ⟨⟩
conv ‵2 ‵2 tt ff df = ⟨⟩
conv (‵Σ A A′) (‵Σ B B′) (a , a′) (b , b′) df with DecUF A B a b
... | ff , _ = ⟨⟩
... | tt , _ with DecUF (A′ a) (B′ b) a′ b′
... | ff , _ = ⟨⟩
conv (‵Σ A A′) (‵Σ B B′) (a , a′) (b , b′) (ff , snd₃) | tt , q | tt , snd₂
  with () ← magic _ _ _ _ q snd₃
conv (‵Σ A A′) (‵Σ B B′) (a , a′) (b , b′) (tt , snd₃) | tt , snd₁ | tt , snd₂
 with () ← magic _ _ _ _ snd₂ snd₃

_+UF_ : UF → UF → UF
S +UF T = ‵Σ ‵2 λ { ff → S ; tt → T }
_×UF_ : UF → UF → UF
S ×UF T = ‵Σ S λ _ → T

-- we can tabulate (i.e. list/enumerate) finite sets, but we can still define
-- aliases of elements of a lambda calculus (which gives us a way of reasoning
-- about terms in UF)

-- "functions" from finite sets
PI : (S : UF)(R : ⟦ S ⟧FE → UE SET) → UE SET
PI (so ff) R = ‵Prf ‵1
PI (so tt) R = R _
PI ‵2 R = ‵Σ (R ff) λ _ → R tt
PI (‵Σ S T) R = PI S λ s → PI (T s) λ t → R (s , t)

-- abstraction/tabulation
LAM : (S : UF){R : ⟦ S ⟧FE → UE SET}
    → ⟦ ‵Π ⟦ S ⟧F R ⟧UE
    → ⟦ PI S R ⟧UE
LAM (so ff) f = _
LAM (so tt) f = f _
LAM ‵2 f = (f ff) , (f tt)
LAM (‵Σ S T) f = LAM S λ s → LAM (T s) λ t → f (s , t)

-- TODO: this shouldn′t be this complex (or even needed at all)
LAM-EXT : (S : UF){R : ⟦ S ⟧FE → UE SET}
        → (f g : ⟦ ‵Π ⟦ S ⟧F R ⟧UE)
        → (q : (s : ⟦ S ⟧FE) → Eq′ (R s) (R s) (f s) (g s))
        → Eq′ (PI S R) (PI S R) (LAM S f) (LAM S g)
LAM-EXT (so ff) f h q = EqR′
LAM-EXT (so tt) f h q = q _
LAM-EXT ‵2 f h q = congUE₂ _,_ (q ff) (q tt)
LAM-EXT (‵Σ S T) f h q = LAM-EXT S (λ s → LAM (T s) (λ t → f (s , t))) (λ s → LAM (T s) (λ t → h (s , t))) λ s′ → LAM-EXT (T s′) (λ t → f (s′ , t)) (λ t → h (s′ , t)) (λ s′′ → q (s′ , s′′))

-- application/projection
APP : (S : UF){R : ⟦ S ⟧FE → UE SET}
    → ⟦ PI S R ⟧UE
    → ⟦ ‵Π ⟦ S ⟧F R ⟧UE
APP (so tt) r _ = r
APP ‵2 (f0 , f1) ff = f0
APP ‵2 (f0 , f1) tt = f1
APP (‵Σ S T) f (s , t) = APP (T s) (APP S f s) t


BETA : (S : UF){R : ⟦ S ⟧FE → UE SET}
    → (f : ⟦ ‵Π ⟦ S ⟧F R ⟧UE)(s : ⟦ S ⟧FE)
    → Eq′ (R s) (R s) (APP S {R} (LAM S f) s) (f s)
BETA (so tt) {R} f s = EqR′
BETA ‵2 {R} f ff = EqR′
BETA ‵2 {R} f tt = EqR′
BETA (‵Σ S T) {R} f (s , t)
  =  (APP (T s) (APP S (LAM S (λ s₁ → LAM (T s₁) (λ t₁ → f (s₁ , t₁)))) s) t)
  =[[ congUE (λ f → APP (T s) f t) (BETA S (λ s′ → LAM (T s′) (λ t → f (s′ , t))) s) ]]>
     APP (T s) (LAM (T s) (λ t′ → f (s , t′))) t
  =[[ BETA (T s) (λ t′ → f (s , t′)) t ]]>
      (f (s , t)) [[QED]]


ETA : (S : UF){R : ⟦ S ⟧FE → UE SET}
    → (f : ⟦ PI S R ⟧UE)
    → Eq′ (PI S R) (PI S R) (LAM S (λ s → APP S f s)) f
ETA (so tt) f = EqR′
ETA (so ff) f = EqR′
ETA ‵2 f = EqR′
ETA (‵Σ S T) f
  =  (LAM S (λ s → LAM (T s) (λ t → APP (T s) (APP S f s) t)))
  =[[ LAM-EXT S (λ s → LAM (T s) (λ t → APP (T s) (APP S f s) t)) (λ s → APP S f s) (λ s → ETA (T s) (APP S f s)) ]]>
     LAM S (λ s → (APP S f s))
  =[[ ETA S f ]]>
     f [[QED]]


-- finite families

--  -- a finite set
--  -- for each element of the finite set Ix, compute its index

record FinFam (M : UE SET) : Set where
  field Ix : UF
        ix : ⟦ Ix ⟧FE → ⟦ M ⟧UE
  -- ^^  on the type level, a function is fine,
  -- vv  but on the value level, we want the tabulation


  -- Pow representation, a pair of the set and a proof that the index is the one I was looking for
  Pow : ⟦ M ⟧UE → UE SET
  Pow m = ‵Σ ⟦ Ix ⟧F λ x → ‵Prf (Eq M M (ix x) m )

open FinFam public

EqFinFam : {M : UE SET} (F G : FinFam M) → UE PROP
EqFinFam {M} record { Ix = IxF ; ix = ixF } record { Ix = IxG ; ix = ixG }
  = Π₂ ⟦ IxF ⟧F ⟦ IxG ⟧F (λ f g → Eq M M (ixF f) (ixG g))


EqF : {M : UE SET} → (F G : FinFam M) → (f : ⟦ Ix F ⟧FE) → (g : ⟦ Ix G ⟧FE) → UE PROP
EqF {M} record { Ix = IxF ; ix = ixF } record { Ix = IxG ; ix = ixG } f g
  = Eq ⟦ IxF ⟧F ⟦ IxG ⟧F f g &P Eq M M (ixF f) (ixG g)

-- subtraction is possible because UF has decidable equality


_─F_ : {M : UE SET} (F : FinFam M) → ⟦ Ix F ⟧FE → FinFam M
Ix (F ─F p) = ‵Σ (Ix F) (λ m → so (not (fst (DecUFh (Ix F) p m))))
ix (F ─F p) (m , _) = ix F m

-- some finite families

Fam0 : {M : UE SET} → FinFam M
Fam0 = record { Ix = so ff ; ix = λ () }

Fam1 : {M : UE SET} → (m : ⟦ M ⟧UE) → FinFam M
Fam1 m = record { Ix = so tt ; ix = λ _ → m }

ΣFam : {M : UE SET}
   → (S : UF) → (⟦ S ⟧FE → FinFam M)
   → FinFam M
Ix (ΣFam S f) = ‵Σ S λ s → Ix (f s)
ix (ΣFam S f) (s , t) = ix (f s) t

_+F_ : {M : UE SET} → FinFam M → FinFam M → FinFam M
Ix (f +F g) = Ix f +UF Ix g
ix (f +F g) (ff , p) = ix f p
ix (f +F g) (tt , p) = ix g p
