module Lib.Fun where

open import Agda.Builtin.Equality


id : forall {l}{X : Set l} → X → X
id x = x

_-_ : forall{i j k}{X : Set i}{Y : X → Set j}{Z : (x : X) → Y x → Set k}
    → (f : (x : X) → Y x) → (g : {x : X}(y : Y x) → Z x y)
    → (x : X) → Z x (f x)
(f - g) x = g (f x)

_∘_ : forall {i j k}{X : Set i}{Y : Set j}{Z : Set k}
    → (Y → Z) → (X → Y)
    → X → Z
(g ∘ f) x = g (f x)


_=^ : ∀ {l}{X : Set l}(x : X) → x ≡ x
x =^ = refl

_=$=_ : forall {k}{l}{S : Set k}{T : Set l}
  {f g : S -> T} -> f ≡ g ->
  {x y : S} -> x ≡ y ->
  f x ≡ g y
refl =$= refl = refl

_$=_ : forall {k}{l}{S : Set k}{T : Set l}
  (f : S -> T) →
  {x y : S} -> x ≡ y ->
  f x ≡ f y
f $= q = f =^ =$= q
infixl 10 _=^ _=$=_ _$=_

cong : ∀ {l}{A B : Set l} (f : A → B) {x y} → x ≡ y → f x ≡ f y
cong = _$=_


cong₂ : ∀ {l}{A B C : Set l}(f : A → B → C) {x y u v} → x ≡ y → u ≡ v → f x u ≡ f y v
cong₂ f xy uv = f $= xy =$= uv


module _ {l}{X : Set l} where

  _=[_]>_ : (x : X){y z : X} → (x ≡ y) → (y ≡ z) → (x ≡ z)
  x =[ refl ]> q = q


  _<[_]=_ : (x : X){y z : X} → (y ≡ x) → (y ≡ z) → (x ≡ z)
  x <[ refl ]= q = q

  infixr 2 _=[_]>_ _<[_]=_

  _[QED] : (x : X) → x ≡ x
  x [QED] = refl

  infixr 3 _[QED]


sym : ∀ {l}{A : Set l} {x y : A} → x ≡ y → y ≡ x
sym refl = refl
