data Desc (I : Set) : Set1 where
  var  : (i : I) → Desc I
  kon  : (A : Set) → Desc I
  sig  : (A : Set) → (A → Desc I) → Desc I
  pair : Desc I → Desc I → Desc I

⟦_⟧D : {I : Set} → Desc I → (I → Set) → Set
⟦ var i ⟧D X = X i
⟦ kon A ⟧D _ = A
⟦ sig A d ⟧D X = Σ A \ a → ⟦ d a ⟧D X
⟦ pair d e ⟧D X = ⟦ d ⟧D X × ⟦ e ⟧D X

t = ⟦ sig Two (\ _ → sig Two var) ⟧D (\ _ → Two)

Sh : {I : Set} → (D : Desc I) → Set
Sh D = ⟦ D ⟧D (\ _ → One)


Pos : {I : Set} → (D : Desc I) → Sh D → I → Set
Pos (var i) dx j = i ≡ j
Pos (kon A) dx i = Zero
Pos (sig A D) (a , dx) i = Pos (D a) dx i
Pos (pair D E) (dx , ex) i = Pos D dx i + Pos E ex i


module _ {I : Set} where

  mapD : (D : Desc I)
       → forall {X Y : I → Set} → ({i : I} → X i → Y i)
       → ⟦ D ⟧D X → ⟦ D ⟧D Y
  mapD (var i) f x = f x
  mapD (kon A) f a = a
  mapD (sig A D) f (a , dx) = a , mapD (D a) f dx
  mapD (pair D E) f (dx , ex) = (mapD D f dx) , (mapD E f ex)



  mapDID : (D : Desc I) {X : I → Set} → (f : ({i : I} → X i → X i))
         → (forall {i} (x : X i) → f x ≡ x) → forall (dx : ⟦ D ⟧D X) → mapD D f dx ≡ dx
  mapDID (var i) f q x = q x
  mapDID (kon A) f q a = refl
  mapDID (sig A D) f q (a , dx) = cong (a ,_) (mapDID (D a) f q dx)
  mapDID (pair D E) f q (dx , ex) = cong₂ _,_ (mapDID D f q dx) (mapDID E f q ex)

  mapDco : (D : Desc I){X Y Z : I → Set}
         → (f : {i : I} → X i → Y i)
         → (g : {i : I} → Y i → Z i)
         → (h : {i : I} → X i → Z i)
         → (forall {i}(x : X i) → h x ≡ g (f x))
         → forall (dx : ⟦ D ⟧D X) → mapD D h dx ≡ mapD D g (mapD D f dx)
  mapDco (var i) f g h q x = q x
  mapDco (kon A) f g h q a = refl
  mapDco (sig A D) f g h q (a , dx) = cong (a ,_) (mapDco (D a) f g h q dx)
  mapDco (pair D E) f g h q (dx , ex) = cong₂ _,_ (mapDco D f g h q dx) (mapDco E f g h q ex)

  mapDcoco : (D : Desc I){X Y Y' Z : I → Set}
         → (f : {i : I} → X i → Y i)
         → (g : {i : I} → Y i → Z i)
         → (f' : {i : I} → X i → Y' i)
         → (g' : {i : I} → Y' i → Z i)
         → (forall {i}(x : X i) → g (f x) ≡ g' (f' x))
         → forall (dx : ⟦ D ⟧D X)
         → mapD D g (mapD D f dx) ≡ mapD D g' (mapD D f' dx)
  mapDcoco (var i) f g f' g' q dx = q _
  mapDcoco (kon A) f g f' g' q dx = refl
  mapDcoco (sig A D) f g f' g' q (a , dx) = cong (a ,_) (mapDcoco (D a) f g f' g' q dx)
  mapDcoco (pair D E) f g f' g' q (dx , ex) = cong₂ _,_ (mapDcoco D f g f' g' q dx) (mapDcoco E f g f' g' q ex)

  retD : I → Desc I
  retD i = var i



  record NT (F G : Desc I) : Set1 where
    field
      transform : {X : I → Set} → ⟦ F ⟧D X → ⟦ G ⟧D X
      naturality : {X Y : I → Set} → (f g : [ X →i Y ])
                 → (forall {i} → (x : X i) → f x ≡ g x)
                 → (fx : ⟦ F ⟧D X)
                 → transform (mapD F f fx) ≡ mapD G g (transform fx)

  open NT public

   {-

  sh D = mapD D _

  proj : forall {X : I → Set} → (D : Desc I) → (dx : ⟦ D ⟧D X) → [ Pos D (sh D dx) →i X ]
  proj (var i) dx refl = dx
  proj (sig A D) (a , dx) p = proj (D a) dx p
  proj (pair D E) (dx , ex) (ff , p) = proj D dx p

         → (F : Desc I)
         → (fx : ⟦ F ⟧D X)
         → (f : [ X →i Y ])
         → sh F (mapD F f fx) ≡ sh F fx

     → ((s : Sh F) → ⟦ G ⟧D (Pos F s))
     → NT F G

module _ {I : Set} where

  -- partial (symbolic differentiation)
  Partial* : Desc I → I → Desc I
  Partial* (var j) i      = kon (i ≡ j)
  Partial* (kon A) i     = kon Zero
  Partial* (sig A D) i   = sig A (λ a → Partial* (D a) i)
  Partial* (pair D E) i  = sig Two (λ  { ff → pair (Partial* D i) E
                                       ; tt → pair D (Partial* E i)})
  -- description of a decomposition
  Pointed* : Desc I → I → Desc I
  Pointed* D i = pair (Partial* D i) (var i)

  Decomposition : (D : Desc I) → (X : I → Set) → I → Set
  Decomposition D X i = ⟦ Pointed* D i ⟧D X

  -- given a decomposition, I can "go up" by plugging the element into the hole
  up* : (D : Desc I) → (X : I → Set) → {i : I}
     → Decomposition D X i → ⟦ D ⟧D X
  up* (var x) X (refl , xi) = xi
  up* (sig A D) X ((a , p) , xi) = a , up* (D a) X (p , xi)
  up* (pair D E) X ((ff , (p , ex)) , xi) = up* D X (p , xi) , ex
  up* (pair D E) X ((tt , (dx , p)) , xi) = dx , up* E X (p , xi)

  -- up* commutes with mapDing a function across the whole structure
  naturalUp* : (D : Desc I) → (X Y : I → Set) → (f : [ X →i Y ]) → {i : I} → (dxx : Decomposition D X i)
            → up* D Y (mapD (Pointed* D i) f dxx) ≡ mapD D f (up* D X dxx)
  naturalUp* (var i) X Y f (refl , x) = refl
  naturalUp* (sig A D) X Y f ((a , dx) , x) = cong (a ,_) (naturalUp* (D a) X Y f (dx , x))
  naturalUp* (pair D E) X Y f ((ff , (dx' , ex)) , x) = cong₂ _,_ (naturalUp* D X Y f (dx' , x)) refl
  naturalUp* (pair D E) X Y f ((tt , (dx , ex')) , x) = cong₂ _,_ refl (naturalUp* E X Y f (ex' , x))

  {-
  -- co-Kleisli version of down
  down' : (D : Desc I) → (X Y : I → Set)
        → (f : [ Decomposition D X →i Y ])
        → ⟦ D ⟧D X → ⟦ D ⟧D Y
  down' (var i) X Y f x = f (refl , x)
  down' (kon A) X Y f a = a
  down' (sig A D) X Y f (a , dx)
    = a , down' (D a) X Y (mapΣ (a ,_) id - f) dx
  down' (pair D E) X Y f (dx , ex)
    = down' D X Y (mapΣ ((_, ex) - (ff ,_)) id - f) dx
    , down' E X Y (mapΣ ((dx ,_) - (tt ,_)) id - f) ex

  -- given a whole thing, I can "go down" by splitting it into a thing with a hole, and an element
  down* : (D : Desc I) → (X : I → Set)
       → ⟦ D ⟧D X → ⟦ D ⟧D (Decomposition D X)
  down* (var i) X x = refl , x
  down* (kon A) X a = a
  down* (sig A D) X (a , dx)  = a , mapD (D a) (mapΣ (a ,_) id) (down* (D a) X dx)
  down* (pair D E) X (dx , ex)
    = mapD D (mapΣ ((_, ex) - (ff ,_)) id) (down* D X dx)
    , mapD E (mapΣ ((dx ,_) - (tt ,_)) id) (down* E X ex)

  -- down* is natural
  naturalDown* : (D : Desc I) → (X Y : I → Set) → (f : [ X →i Y ]) → {i : I} → (dx : ⟦ D ⟧D X)
              → down* D Y (mapD D f dx) ≡ mapD D (mapD (Pointed* D _) f) (down* D X dx)
  naturalDown* (var i) X Y f dx = refl
  naturalDown* (kon A) X Y f dx = refl
  naturalDown* (sig A D) X Y f {i} (a , dx) = cong (a ,_)
    (mapD (D a) (mapΣ (a ,_) id) (down* (D a) Y (mapD (D a) f dx))
       <[ sym (cong (mapD (D a) (mapΣ (a ,_) id)) (naturalDown* (D a) X Y f {i} dx)) ]=
     mapD (D a) (mapΣ (a ,_) id) (mapD (D a) (mapD (Pointed* (D a) _) f) (down* (D a) X dx))
       =[ mapDcoco (D a)
                   (mapD (Pointed* (D a) _) f) (mapΣ (a ,_) id)
                   (mapΣ (a ,_) id) (mapD (Pointed* (sig A D) _) f)
                   (\ _ → refl) (down* (D a) X dx) ]>
     mapD (D a) (mapD (Pointed* (sig A D) _) f) (mapD (D a) (mapΣ (a ,_) id) (down* (D a) X dx))
       [QED])
  naturalDown* (pair D E) X Y f {i} (dx , ex) = cong₂ _,_
    -- left side
    (mapD D (mapΣ ((_, mapD E f ex) - (ff ,_)) id)(down* D Y (mapD D f dx))
       =[ cong (mapD D (mapΣ (λ z → ((_, mapD E f ex) - (ff ,_)) z) id)) (naturalDown* D X Y f {i} dx) ]>
     mapD D (mapΣ (λ z → ((_, mapD E f ex) - (ff ,_)) z) id) (mapD D (mapD (Pointed* D _) f) (down* D X dx))
        =[ mapDcoco D
                    (mapD (Pointed* D _) f)
                    (mapΣ (λ z → ((_, mapD E f ex) - (ff ,_)) z) id)
                    (mapΣ (λ z → ((_, ex) - (ff ,_)) z) id)
                    (mapD (Pointed* (pair D E) _) f)
                    (\ _ → refl) (down* D X dx) ]>
     mapD D (mapD (Pointed* (pair D E) _) f) (mapD D (mapΣ ((_, ex) - (ff ,_)) id) (down* D X dx))
       [QED])

    -- right side
    (mapD E (mapΣ ((mapD D f dx ,_) - (tt ,_)) id) (down* E Y (mapD E f ex))
       =[ cong (mapD E (mapΣ ((mapD D f dx ,_) - (tt ,_)) id)) (naturalDown* E X Y f {i} ex) ]>
     mapD E (mapΣ ((mapD D f dx ,_) - (tt ,_)) id) (mapD E (mapD (Pointed* E _) f) (down* E X ex))
       =[ mapDcoco E
                    (mapD (Pointed* E _) f)
                    (mapΣ (λ z → ((mapD D f dx ,_) - (tt ,_)) z) id)
                    (mapΣ (λ z → ((dx ,_) - (tt ,_)) z) id)
                    (mapD (Pointed* (pair D E) _) f)
                    (\_ → refl)
                    (down* E X ex) ]>
     mapD E (mapD (Pointed* (pair D E) _) f) (mapD E (mapΣ ((dx ,_) - (tt ,_)) id) (down* E X ex))
       [QED])


  mapSndDown* : (D : Desc I) → (X : I → Set) → (dx : ⟦ D ⟧D X)
             → mapD D snd (down* D X dx) ≡ dx
  mapSndDown* (var i) X x = refl
  mapSndDown* (kon A) X a = refl
  mapSndDown* (sig A D) X (a , dx) = cong (a ,_)
    (mapD (D (fst (down* (sig A D) X (a , dx)))) snd (snd (down* (sig A D) X (a , dx)))
       <[ mapDco (D a) (mapΣ (a ,_) id) snd snd (\ _ → refl) (down* (D a) X dx) ]=
     mapD (D a) snd (down* (D a) X dx)
       =[ mapSndDown* (D a) X dx ]>
     dx
       [QED])
  mapSndDown* (pair D E) X (dx , ex) = cong₂ _,_
    (mapD D snd (mapD D (mapΣ ((_, ex) - (ff ,_)) id) (down* D X dx))
        <[ mapDco D (mapΣ (\ z → ff , z , ex) id) snd snd (\ _ → refl) (down* D X dx) ]=
      mapD D snd (down* D X dx)
        =[ mapSndDown* D X dx ]>
      dx
        [QED])
    (mapD E snd (snd (down* (pair D E) X (dx , ex)))
       <[ mapDco E (mapΣ (\ z → tt , dx , z) id) snd snd (\ _ → refl) (down* E X ex) ]=
     mapD E snd (down* E X ex)
       =[ mapSndDown* E X ex ]>
     ex
       [QED])

  mapUpDown* : (D : Desc I) → (X : I → Set) → (dx : ⟦ D ⟧D X)
            → mapD D (up* D X) (down* D X dx) ≡ mapD D (\ _ → dx) dx
  mapUpDown* (var i) X x = refl
  mapUpDown* (kon A) X a = refl
  mapUpDown* (sig A D) X (a , dx) = cong (a ,_)
    (mapD (D a) (up* (sig A D) X) (mapD (D a) (mapΣ (a ,_) id) (down* (D a) X dx))
       <[ mapDcoco (D a) (up* (D a) X) (a ,_) (mapΣ (a ,_) id) (up* (sig A D) X) (\ _ → refl) (down* (D a) X dx)  ]=
     mapD (D a) (a ,_) (mapD (D a) (up* (D a) X) (down* (D a) X dx))
       =[ cong (mapD (D a) (a ,_)) (mapUpDown* (D a) X dx) ]>
     mapD (D a) (a ,_)  (mapD (D a) (λ _ → dx) dx)
       <[ mapDco (D a) (λ _ → dx) (a ,_) _ (\ _ → refl) dx ]=
     mapD (D a) (λ _ → a , dx) dx [QED])
  mapUpDown* (pair D E) X (dx , ex) = cong₂ _,_
    (mapD D (up* (pair D E) X) (mapD D (mapΣ (\ x → ff , x , ex) id) (down* D X dx))
       =[ mapDcoco D (mapΣ (λ x → ff , x , ex) id) (up* (pair D E) X) (up* D X) (_, ex) (\ _ → refl) (down* D X dx) ]>
     mapD D (_, ex) (mapD D (up* D X) (down* D X dx))
       =[ cong (mapD D (_, ex)) (mapUpDown* D X dx) ]>
     mapD D (_, ex) (mapD D (λ _ → dx) dx)
       <[ mapDco D (\ _ → dx) (_, ex) _ (\ _ → refl) dx ]=
     mapD D (\ _ → dx , ex) dx
       [QED])
    (mapD E (up* (pair D E) X) (mapD E (mapΣ ((dx ,_) - (tt ,_)) id) (down* E X ex))
       =[ mapDcoco E (mapΣ ((dx ,_) - (tt ,_)) id) (up* (pair D E) X) (up* E X) (dx ,_) (\ _ → refl) (down* E X ex) ]>
     mapD E (dx ,_) (mapD E (up* E X) (down* E X ex))
       =[ cong (mapD E (dx ,_)) (mapUpDown* E X ex) ]>
     mapD E (dx ,_) (mapD E (λ _ → ex) ex)
       <[ mapDco E (\ _ → ex) (dx ,_) _ (\ _ → refl) ex ]=
     mapD E (λ _ → dx , ex) ex
       [QED])

  elsewhere* : (D : Desc I) → (X : I → Set)
            → {i : I} → Decomposition D X i
            → ⟦ Partial* D i ⟧D (Decomposition D X)
  elsewhere* (var i) X (q , x) = q
  elsewhere* (sig A D) X {i} ((a , dx') , x)
    = a , mapD (Partial* (D a) i) (mapΣ (a ,_) id) (elsewhere* (D a) X (dx' , x))
  elsewhere* (pair D E) X {i} ((ff , dx' , ex) , x)
    = ff , mapD (Partial* D i) (mapΣ ((_, ex) - (ff ,_)) id) (elsewhere* D X (dx' , x))
         , mapD E (\ (ex' , y) → (tt , up* D X (dx' , x) , ex') , y) (down* E X ex)
  elsewhere* (pair D E) X {i} ((tt , dx , ex') , x)
    = tt , mapD D (\ (dx' , y) → (ff , dx' , up* E X (ex' , x)) , y) (down* D X dx)
         , mapD (Partial* E i) (mapΣ ((dx ,_) - (tt ,_)) id) (elsewhere* E X (ex' , x))


  elsewhereNatural : (D : Desc I) → (X Y : I → Set)(f : [ X →i Y ]) → {i : I} → (dxx : Decomposition D X i)
                   → elsewhere* D Y (mapD (Pointed* D i) f dxx) ≡ mapD (Partial* D i) (mapD (Pointed* D _) f) (elsewhere* D X dxx)
  elsewhereNatural (var i) X Y f dxx = refl
  elsewhereNatural (sig A D) X Y f {i} ((a , dx) , x) = cong (a ,_)
    (mapD (Partial* (D a) i) (mapΣ (a ,_) id) (elsewhere* (D a) Y (mapD (Partial* (D a) i) f dx , f x))
       =[ cong (mapD (Partial* (D a) i) (mapΣ (a ,_) id)) (elsewhereNatural (D a) X Y f (dx , x)) ]>
     mapD (Partial* (D a) i) (mapΣ (a ,_) id) (mapD (Partial* (D a) i) (mapD (Pointed* (D a) _) f) (elsewhere* (D a) X (dx , x)))
       =[ mapDcoco (Partial* (D a) i) (mapD (Pointed* (D a) _) f) (mapΣ (a ,_) id) (mapΣ (a ,_) id) (mapD (Pointed* (sig A D) _) f) (\ _ → refl) (elsewhere* (D a) X (dx , x)) ]>
     mapD (Partial* (D a) i) (mapD (Pointed* (sig A D) _) f) (mapD (Partial* (D a) i) (mapΣ (a ,_) id) (elsewhere* (D a) X (dx , x)))
       [QED])
  elsewhereNatural (pair D E) X Y f {i} ((ff , dx' , ex) , x) = cong (ff ,_) (cong₂ _,_
    (mapD (Partial* D i) (mapΣ ((_, mapD E f ex) - (ff ,_)) id) (elsewhere* D Y (mapD (Partial* D i) f dx' , f x))
      =[ cong (mapD (Partial* D i) (mapΣ (\ z → ff , z , mapD E f ex) id)) (elsewhereNatural D X Y f (dx' , x)) ]>
    mapD (Partial* D i) (mapΣ (\ z → ff , z , mapD E f ex) id) (mapD (Partial* D i) (mapD (Pointed* D _) f) (elsewhere* D X (dx' , x)))
      =[ mapDcoco (Partial* D i)
                  (mapD (Pointed* D _) f) (mapΣ (λ z → ff , z , mapD E f ex) id)
                  (mapΣ (λ z → ((_, ex) - (λ section → ff , section)) z) id) (mapD (Pointed* (pair D E) _) f)
                  (\ _ → refl) (elsewhere* D X (dx' , x)) ]>
    mapD (Partial* D i) (mapD (Pointed* (pair D E) _) f) (mapD (Partial* D i) (mapΣ ((_, ex) - (\ section → ff , section)) id) (elsewhere* D X (dx' , x)))
      [QED])
    (mapD E (\ (ex , y) → (tt ,  up* D Y (mapD (Partial* D i) f dx' , f x) , ex) , y) (down* E Y (mapD E f ex))
       =[ cong (mapD E (\ (ex , y) → (tt ,  up* D Y (mapD (Partial* D i) f dx' , f x) , ex) , y)) (naturalDown* E X Y f {i} ex) ]>
    (mapD E (\ (ex , y) → (tt ,  up* D Y (mapD (Partial* D i) f dx' , f x) , ex) , y)) (mapD E (mapD (Pointed* E _) f) (down* E X ex))
       =[ sym (mapDco E (mapD (Pointed* E _) f) (\ (ex , y) → (tt , up* D Y (mapD (Partial* D i) f dx' , f x) , ex) , y) _ (\ _ → refl) (down* E X ex)) ]>
     mapD E (\ z → (tt , up* D Y (mapD (Partial* D i) f dx' , f x) , fst (mapD (Pointed* E _) f z)) , snd (mapD (Pointed* E _) f z)) (down* E X ex)
       =[ cong (\ var →  mapD E (\ (ex , y) → (tt , var , mapD (Partial* E _) f ex) , f y) (down* E X ex)) (naturalUp* D X Y f (dx' , x)) ]>
     mapD E (\ (ex , y) → (tt , mapD D f (up* D X (dx' , x)) , mapD (Partial* E _) f ex) , f y) (down* E X ex)

  mapSndElsewhere* : (D : Desc I) → (X : I → Set) → {i : I} → (dx' : ⟦ Partial* D i ⟧D X) → (x : X i)
